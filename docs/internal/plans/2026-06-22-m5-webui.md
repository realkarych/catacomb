# M5 — Embedded Web UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Embed a zero-install vanilla-JS web UI in the `catacomb` binary that renders the live execution graph from the SSE feed via four views (live DAG, timeline, node inspector, run list).

**Architecture:** Static assets (`web/index.html`, `web/app.js`, `web/style.css`) are embedded via `//go:embed` in a `webui` package, served openly (no auth) at `GET /` by the daemon mux. The SSE feed (`GET /v1/subscribe`) gains a new auth helper `authedAllowQuery` that accepts the bearer token via either `Authorization: Bearer <t>` header or `?token=<t>` query parameter (both constant-time-compared), enabling the browser's `EventSource` API. A new `catacomb ui` command reads the discovery file, composes `http://<addr>/?token=<token>`, prints it, and opens the default browser via an injectable seam.

**Tech Stack:** Go 1.26, `net/http` stdlib, `crypto/subtle`, `embed` stdlib, `io/fs` stdlib, `runtime` GOOS-dispatch for the browser opener, vanilla JS + SVG (no third-party library, no build step), `github.com/spf13/cobra` (already in tree).

## Global Constraints

Go 1.26 pure-Go (no cgo); **NO Go comments except `//go:build|//go:embed|//go:generate`** (the `//go:embed` directive is explicitly allowed); **100% line coverage under `-race`** for the Go code (the embedded frontend files are data, not statements — they do not affect Go coverage; the serve handler + auth + cmd ARE covered); golangci-lint v2 clean (gofumpt, goimports, govet shadow, forbidigo bans `time.Sleep`, unparam, errcheck, bodyclose); **never `go mod tidy`** (M5 adds NO Go deps and no third-party JS — the renderer is hand-rolled vanilla JS + SVG); cross-platform (`GOOS=windows go build ./...` clean — the browser-opener seam's OS branch must build on all platforms; tests inject a fake opener); loopback + bearer trust boundary preserved (static open, data gated, token via URL documented); commit per task; never commit to master mid-plan.

```go
func (d *Daemon) authedAllowQuery(token string, next http.HandlerFunc) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		headerOK := subtle.ConstantTimeCompare([]byte(r.Header.Get("Authorization")), []byte("Bearer "+token))
		queryOK := subtle.ConstantTimeCompare([]byte(r.URL.Query().Get("token")), []byte(token))
		if headerOK != 1 && queryOK != 1 {
			w.WriteHeader(http.StatusUnauthorized)
			return
		}
		next(w, r)
	}
}
```

```js
(function () {
  'use strict';

  var params = new URLSearchParams(window.location.search);
  var TOKEN = params.get('token') || '';

  var nodes = new Map();
  var edges = new Map();
  var runs = new Map();
  var selectedRunID = null;
  var selectedNodeID = null;
  var activeView = 'graph';

  function applyDelta(ev) {
    switch (ev.kind) {
      case 'node_upsert':
        if (ev.node) {
          var existing = nodes.get(ev.node.id);
          if (existing && existing.rev >= ev.rev) return;
          nodes.set(ev.node.id, ev.node);
          ensureRun(ev.node.run_id, ev.node.status);
        }
        break;
      case 'node_status':
        if (ev.node) {
          var n = nodes.get(ev.node.id);
          if (!n) { nodes.set(ev.node.id, ev.node); }
          else if (n.rev < ev.rev) { n.status = ev.node.status; n.rev = ev.rev; n.t_end = ev.node.t_end; n.duration_ms = ev.node.duration_ms; }
          ensureRun(ev.node.run_id, ev.node.status);
        }
        break;
      case 'node_merge':
        if (ev.node && ev.old_id) {
          var old = nodes.get(ev.old_id);
          if (old) nodes.delete(ev.old_id);
          nodes.set(ev.node.id, ev.node);
        }
        break;
      case 'edge_upsert':
        if (ev.edge) {
          var ee = edges.get(ev.edge.id);
          if (ee && ee.rev >= ev.rev) return;
          edges.set(ev.edge.id, ev.edge);
        }
        break;
      case 'edge_delete':
        if (ev.edge) edges.delete(ev.edge.id);
        break;
      case 'run_start':
        ensureRun(ev.run_id, 'running');
        break;
      case 'run_end':
        var r = runs.get(ev.run_id);
        if (r) r.status = ev.node ? ev.node.status : 'ok';
        break;
      default:
        break;
    }
    render();
  }

  function ensureRun(runID, status) {
    if (!runID) return;
    if (!runs.has(runID)) runs.set(runID, { id: runID, status: status || 'pending' });
    else if (status === 'running') runs.get(runID).status = 'running';
  }

  function bfsLayout() {
    var layers = [];
    var visited = new Set();
    var nodeList = Array.from(nodes.values());
    if (selectedRunID) nodeList = nodeList.filter(function (n) { return n.run_id === selectedRunID; });

    var childSet = new Set();
    var edgeList = Array.from(edges.values());
    edgeList.forEach(function (e) {
      if (e.type === 'parent_child') childSet.add(e.dst);
    });

    var roots = nodeList.filter(function (n) { return !childSet.has(n.id); });
    if (roots.length === 0 && nodeList.length > 0) roots = [nodeList[0]];

    function bfs(root) {
      var queue = [{ node: root, depth: 0 }];
      while (queue.length) {
        var item = queue.shift();
        var nd = item.node;
        var depth = item.depth;
        if (visited.has(nd.id)) continue;
        visited.add(nd.id);
        while (layers.length <= depth) layers.push([]);
        layers[depth].push(nd);
        var children = edgeList
          .filter(function (e) { return e.type === 'parent_child' && e.src === nd.id; })
          .map(function (e) { return nodes.get(e.dst); })
          .filter(Boolean);
        children.forEach(function (c) { queue.push({ node: c, depth: depth + 1 }); });
      }
    }

    roots.forEach(bfs);
    nodeList.forEach(function (n) {
      if (!visited.has(n.id)) {
        while (layers.length <= layers.length) layers.push([]);
        if (!layers[layers.length - 1]) layers.push([]);
        layers[layers.length - 1].push(n);
        visited.add(n.id);
      }
    });

    var W = 140, H = 56, PX = 60, PY = 40;
    var positions = new Map();
    layers.forEach(function (layer, col) {
      layer.forEach(function (nd, row) {
        positions.set(nd.id, { x: PX + col * (W + PX), y: PY + row * (H + PY) });
      });
    });
    return positions;
  }

  function renderGraph() {
    var svg = document.getElementById('graph-svg');
    var W = 140, H = 46;
    var positions = bfsLayout();
    var nodeList = Array.from(nodes.values());
    if (selectedRunID) nodeList = nodeList.filter(function (n) { return n.run_id === selectedRunID; });
    var edgeList = Array.from(edges.values()).filter(function (e) {
      return positions.has(e.src) && positions.has(e.dst);
    });

    var svgNS = 'http://www.w3.org/2000/svg';
    var maxX = 0, maxY = 0;
    positions.forEach(function (p) {
      if (p.x + W + 60 > maxX) maxX = p.x + W + 60;
      if (p.y + H + 40 > maxY) maxY = p.y + H + 40;
    });

    var markup = '<svg xmlns="' + svgNS + '" width="' + Math.max(maxX, 400) + '" height="' + Math.max(maxY, 300) + '">';

    edgeList.forEach(function (e) {
      var s = positions.get(e.src), d = positions.get(e.dst);
      var x1 = s.x + W, y1 = s.y + H / 2, x2 = d.x, y2 = d.y + H / 2;
      var mx = (x1 + x2) / 2;
      markup += '<path class="edge-line" d="M' + x1 + ',' + y1 + ' C' + mx + ',' + y1 + ' ' + mx + ',' + y2 + ' ' + x2 + ',' + y2 + '"/>';
    });

    nodeList.forEach(function (nd) {
      var p = positions.get(nd.id);
      if (!p) return;
      var tc = 'node-' + (nd.type || 'marker');
      var sc = 'status-' + (nd.status || 'pending');
      var sel = nd.id === selectedNodeID ? ' stroke-width="3"' : '';
      var label = (nd.name || nd.type || nd.id).slice(0, 18);
      markup += '<g class="node-group" data-id="' + nd.id + '">';
      markup += '<rect class="node-rect ' + tc + ' ' + sc + '" x="' + p.x + '" y="' + p.y + '" width="' + W + '" height="' + H + '"' + sel + '/>';
      markup += '<text class="node-label" x="' + (p.x + 6) + '" y="' + (p.y + 16) + '">' + escapeXML(label) + '</text>';
      markup += '<text class="node-label" style="fill:#8b949e;font-size:9px" x="' + (p.x + 6) + '" y="' + (p.y + 30) + '">' + escapeXML(nd.status || '') + '</text>';
      markup += '</g>';
    });

    markup += '</svg>';
    svg.outerHTML = markup;

    document.getElementById('view-graph').querySelectorAll('.node-group').forEach(function (g) {
      g.addEventListener('click', function () {
        selectedNodeID = g.getAttribute('data-id');
        renderInspector(nodes.get(selectedNodeID));
        render();
      });
    });
  }

  function renderTimeline() {
    var svg = document.getElementById('timeline-svg');
    var svgNS = 'http://www.w3.org/2000/svg';
    var runList = Array.from(nodes.values()).filter(function (n) { return n.t_start; });
    if (runList.length === 0) { svg.innerHTML = '<text x="20" y="30" fill="#484f58">No timing data yet</text>'; return; }

    var minT = Infinity, maxT = -Infinity;
    runList.forEach(function (n) {
      var s = new Date(n.t_start).getTime();
      var e = n.t_end ? new Date(n.t_end).getTime() : Date.now();
      if (s < minT) minT = s;
      if (e > maxT) maxT = e;
    });

    var totalMs = maxT - minT || 1;
    var W = 800, rowH = 24, PY = 30, PX = 150;
    var viewW = W + PX + 20, viewH = PY + runList.length * (rowH + 4) + 20;

    var markup = '<svg xmlns="' + svgNS + '" width="' + viewW + '" height="' + viewH + '">';
    runList.forEach(function (n, i) {
      var s = new Date(n.t_start).getTime();
      var e = n.t_end ? new Date(n.t_end).getTime() : Date.now();
      var x = PX + ((s - minT) / totalMs) * W;
      var w = Math.max(((e - s) / totalMs) * W, 4);
      var y = PY + i * (rowH + 4);
      var tc = 'node-' + (n.type || 'marker');
      var label = (n.name || n.id).slice(0, 20);
      markup += '<rect class="timeline-bar ' + tc + '" x="' + x + '" y="' + y + '" width="' + w + '" height="' + rowH + '"/>';
      markup += '<text class="timeline-label" x="4" y="' + (y + 16) + '">' + escapeXML(label) + '</text>';
    });
    markup += '</svg>';
    svg.outerHTML = markup;
  }

  function renderInspector(nd) {
    var panel = document.getElementById('inspector-panel');
    if (!nd) { panel.innerHTML = '<p class="hint">Click a node in the Graph view</p>'; return; }
    var rows = [
      ['id', nd.id], ['type', nd.type], ['name', nd.name || ''], ['status', nd.status || ''],
      ['run_id', nd.run_id || ''], ['t_start', nd.t_start || ''], ['t_end', nd.t_end || ''],
      ['duration_ms', nd.duration_ms != null ? nd.duration_ms : ''],
      ['tokens_in', nd.tokens_in != null ? nd.tokens_in : ''],
      ['tokens_out', nd.tokens_out != null ? nd.tokens_out : ''],
      ['cost_usd', nd.cost_usd != null ? nd.cost_usd.toFixed(6) : ''],
      ['payload_hash', nd.payload_hash || ''],
      ['tier', nd.tier || ''],
    ];
    if (nd.attrs) {
      Object.keys(nd.attrs).forEach(function (k) {
        rows.push(['attr:' + k, JSON.stringify(nd.attrs[k])]);
      });
    }
    if (nd.sources && nd.sources.length) {
      nd.sources.forEach(function (s, i) {
        rows.push(['source[' + i + ']', s.source + ' / ' + s.obs_id]);
      });
    }
    var html = '<h3>' + escapeHTML(nd.name || nd.type || nd.id) + '</h3><table>';
    rows.forEach(function (row) {
      if (row[1] === '' || row[1] == null) return;
      html += '<tr><td>' + escapeHTML(row[0]) + '</td><td>' + escapeHTML(String(row[1])) + '</td></tr>';
    });
    html += '</table>';
    panel.innerHTML = html;
  }

  function renderRuns() {
    var filterVal = (document.getElementById('runs-filter') || {}).value || '';
    var list = document.getElementById('runs-list');
    if (!list) return;
    var runArr = Array.from(runs.values()).filter(function (r) {
      return !filterVal || r.id.includes(filterVal);
    });
    var html = '';
    runArr.forEach(function (r) {
      var sel = r.id === selectedRunID ? ' selected' : '';
      var sc = 'run-status-' + (r.status || 'pending');
      html += '<li data-run="' + r.id + '" class="' + sel + '"><span class="run-status ' + sc + '"></span>' + escapeHTML(r.id.slice(0, 32)) + '</li>';
    });
    list.innerHTML = html;
    list.querySelectorAll('li').forEach(function (li) {
      li.addEventListener('click', function () {
        selectedRunID = li.getAttribute('data-run');
        render();
      });
    });
  }

  function render() {
    if (activeView === 'graph') renderGraph();
    else if (activeView === 'timeline') renderTimeline();
    else if (activeView === 'inspector') renderInspector(selectedNodeID ? nodes.get(selectedNodeID) : null);
    else if (activeView === 'runs') renderRuns();
  }

  function escapeXML(s) {
    return String(s).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;');
  }

  function escapeHTML(s) { return escapeXML(s); }

  document.querySelectorAll('.tab').forEach(function (btn) {
    btn.addEventListener('click', function () {

      activeView = btn.getAttribute('data-view');
      var view = document.getElementById('view-' + activeView);

  });

  var filterInput = document.getElementById('runs-filter');
  if (filterInput) {
    filterInput.addEventListener('input', function () { renderRuns(); });
  }

  var connStatus = document.getElementById('conn-status');

  function connect() {
    var es = new EventSource('/v1/subscribe?token=' + encodeURIComponent(TOKEN));
    es.onopen = function () {
      if (connStatus) { connStatus.textContent = 'connected'; connStatus.className = 'connected'; }
    };
    es.onerror = function () {
      if (connStatus) { connStatus.textContent = 'reconnecting…'; connStatus.className = 'error'; }
    };
    es.onmessage = function (e) {
      try {
        var ev = JSON.parse(e.data);
        applyDelta(ev);
      } catch (_) {}
    };
  }

  connect();
  render();
```

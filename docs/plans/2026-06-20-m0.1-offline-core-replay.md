# Catacomb M0.1 — Offline Core & Replay — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the transport-agnostic core that turns a recorded Claude Code session (transcript JSONL) into a canonical execution graph, persists it to SQLite, and exports it to JSONL — all offline and unit-testable.

**Architecture:** A reusable core (`model`, `ingest/jsonl`, `reduce`, `store`, `export/jsonl`) plus a thin `cmd/catacomb` cobra CLI. Data flows transcript file → JSONL parser → `[]model.Observation` → reducer (canonical-entity merge) → in-memory `reduce.Graph` → SQLite store (write-through) and JSONL exporter. The reducer is a deterministic reduction of an observation stream, so the same fixtures always yield the same golden graph — the basis of every test.

**Tech Stack:** Go 1.26, `github.com/spf13/cobra` (CLI), `modernc.org/sqlite` (pure-Go SQLite, no cgo), `github.com/stretchr/testify` (assertions).

## Global Constraints

These apply to **every** task; they are the repo's standing rules (see `AGENTS.md`), repeated here because each task is implemented in isolation.

- **Go 1.26+**; module `github.com/realkarych/catacomb`; **pure-Go, no cgo** (SQLite via `modernc.org/sqlite`, never `mattn/go-sqlite3`).
- **No comments in any Go code** — none, not even doc comments. Only `//go:build|embed|generate` directives are allowed; `internal/codepolicy` fails the build on any other comment. Every snippet below is already comment-free; keep it that way. Put explanation in identifiers, not comments.
- **testify** is the test stack: `require` for fatal assertions, `assert` for continuing ones. Table-driven where natural. No `time.Sleep` (forbidigo).
- **100% line coverage** is enforced by `make cover` (`.testcoverage.yml`, file/package/total = 100). Each task below ends with a **Coverage closure** list naming every branch its tests must hit. Untestable code is a refactoring signal — design seams, do not add exclusions.
- **Core packages** (`model`, `ingest/*`, `reduce`, `store`, `export/*`) import no CLI/daemon/network code.
- **Workflow:** all work happens on the branch created in Task 0; commit per task on that branch; open **one** squash PR at the end. No direct commits to `master`.
- The repo scaffold already exists on `master` (go.mod, LICENSE, README, Makefile, CI, linters, `internal/codepolicy`, and a placeholder `cmd/catacomb/main.go`). **Do not recreate these.** This plan only adds packages and replaces the placeholder `main.go`.

```go
import (
```
```go
	"crypto/sha256"
	"encoding/hex"
```

```go
func HashPayload(p *Payload) string {
	if p == nil {
		return ""
	}
	h := sha256.New()
	h.Write(p.Input)
	h.Write(p.Output)
	return hex.EncodeToString(h.Sum(nil))
}
```

Create `ingest/jsonl/jsonl.go`:

```go
package jsonl

import (
	"bufio"
	"encoding/json"
	"fmt"
	"io"
	"strconv"
	"strings"
	"time"

	"github.com/realkarych/catacomb/model"
)

type line struct {
	Type      string          `json:"type"`
	UUID      string          `json:"uuid"`
	SessionID string          `json:"sessionId"`
	Timestamp string          `json:"timestamp"`
	Message   json.RawMessage `json:"message"`
}

type message struct {
	Role    string          `json:"role"`
	ID      string          `json:"id"`
	Model   string          `json:"model"`
	Content json.RawMessage `json:"content"`
	Usage   *usage          `json:"usage"`
}

type usage struct {
	InputTokens  int64 `json:"input_tokens"`
	OutputTokens int64 `json:"output_tokens"`
}

type block struct {
	Type      string          `json:"type"`
	ID        string          `json:"id"`
	Name      string          `json:"name"`
	Input     json.RawMessage `json:"input"`
	ToolUseID string          `json:"tool_use_id"`
	Content   json.RawMessage `json:"content"`
	IsError   bool            `json:"is_error"`
}

type partial struct {
	kind        string
	correlation model.Correlation
	attrs       map[string]any
	payload     *model.Payload
}

func ParseReader(r io.Reader) ([]model.Observation, error) {
	sc := bufio.NewScanner(r)
	sc.Buffer(make([]byte, 0, 1024*1024), 16*1024*1024)

	var out []model.Observation
	var seq uint64
	for sc.Scan() {
		raw := strings.TrimSpace(sc.Text())
		if raw == "" {
			continue
		}
		ln, parts, err := decodeLine([]byte(raw))
		if err != nil {
			return nil, err
		}
		ts, _ := time.Parse(time.RFC3339, ln.Timestamp)
		for _, p := range parts {
			out = append(out, model.Observation{
				ObsID:       "obs-" + strconv.FormatUint(seq, 10),
				RunID:       ln.SessionID,
				Source:      model.SourceJSONL,
				Kind:        p.kind,
				Correlation: p.correlation,
				Attrs:       p.attrs,
				Payload:     p.payload,
				ObservedAt:  ts,
				Seq:         seq,
			})
			seq++
		}
	}
	if err := sc.Err(); err != nil {
		return nil, fmt.Errorf("jsonl.ParseReader: %w", err)
	}
	return out, nil
}

func decodeLine(raw []byte) (line, []partial, error) {
	var ln line
	if err := json.Unmarshal(raw, &ln); err != nil {
		return ln, nil, fmt.Errorf("jsonl.decodeLine: %w", err)
	}
	if len(ln.Message) == 0 {
		return ln, nil, nil
	}
	var msg message
	if err := json.Unmarshal(ln.Message, &msg); err != nil {
		return ln, nil, fmt.Errorf("jsonl.decodeLine.message: %w", err)
	}
	text, blocks, err := decodeContent(msg.Content)
	if err != nil {
		return ln, nil, err
	}
	base := model.Correlation{SessionID: ln.SessionID, UUID: ln.UUID}
	switch ln.Type {
	case "user":
		return ln, userParts(base, text, blocks), nil
	case "assistant":
		return ln, assistantParts(base, msg, blocks), nil
	default:
		return ln, nil, nil
	}
}

func decodeContent(raw json.RawMessage) (string, []block, error) {
	var text string
	if err := json.Unmarshal(raw, &text); err == nil {
		return text, nil, nil
	}
	var blocks []block
	if err := json.Unmarshal(raw, &blocks); err != nil {
		return "", nil, fmt.Errorf("jsonl.decodeContent: %w", err)
	}
	return "", blocks, nil
}

func userParts(base model.Correlation, text string, blocks []block) []partial {
	var parts []partial
	if text != "" {
		parts = append(parts, partial{kind: "user_prompt", correlation: base})
	}
	for _, b := range blocks {
		if b.Type != "tool_result" {
			continue
		}
		status := model.StatusOK
		if b.IsError {
			status = model.StatusError
		}
		c := base
		c.ToolUseID = b.ToolUseID
		pl := &model.Payload{Output: b.Content}
		pl.Hash = model.HashPayload(pl)
		parts = append(parts, partial{
			kind:        "tool_result",
			correlation: c,
			attrs:       map[string]any{"status": string(status)},
			payload:     pl,
		})
	}
	return parts
}

func assistantParts(base model.Correlation, msg message, blocks []block) []partial {
	turn := base
	turn.MessageID = msg.ID
	attrs := map[string]any{"model": msg.Model}
	if msg.Usage != nil {
		attrs["tokens_in"] = msg.Usage.InputTokens
		attrs["tokens_out"] = msg.Usage.OutputTokens
	}
	parts := []partial{{kind: "assistant_turn", correlation: turn, attrs: attrs}}
```

```go
package reduce

import "github.com/realkarych/catacomb/model"

type Graph struct {
	Nodes map[string]*model.Node
	Edges map[string]*model.Edge
}

func NewGraph() *Graph {
	return &Graph{Nodes: map[string]*model.Node{}, Edges: map[string]*model.Edge{}}
}

func (g *Graph) node(id, runID string, t model.NodeType) *model.Node {
	n, ok := g.Nodes[id]
	if !ok {
		n = &model.Node{ID: id, RunID: runID, Type: t, Tier: "core"}
		g.Nodes[id] = n
	}
	return n
}

func (g *Graph) upsertEdge(runID string, t model.EdgeType, src, dst string) {
	if src == "" || dst == "" {
		return
	}
	id := model.EdgeID(runID, t, src, dst)
	if _, ok := g.Edges[id]; !ok {
		g.Edges[id] = &model.Edge{ID: id, RunID: runID, Type: t, Src: src, Dst: dst}
	}
}
```

```go
package reduce

import (
	"strings"

	"github.com/realkarych/catacomb/model"
)

func (g *Graph) ApplyAll(obs []model.Observation) {
	for _, o := range obs {
		g.Apply(o)
	}
}

func (g *Graph) Apply(o model.Observation) {
	g.node(model.SessionNodeID(o.RunID), o.RunID, model.NodeSession)
	switch o.Kind {
	case "user_prompt":
		n := g.node(model.UserPromptID(o.RunID, o.Correlation.UUID), o.RunID, model.NodeUserPrompt)
		g.stamp(n, o)
		g.upsertEdge(o.RunID, model.EdgeParentChild, model.SessionNodeID(o.RunID), n.ID)
	case "assistant_turn":
		n := g.node(model.AssistantTurnID(o.RunID, o.Correlation.MessageID), o.RunID, model.NodeAssistantTurn)
		g.stamp(n, o)
		applyTokens(n, o.Attrs)
	case "assistant_tool_use", "tool_result":
		g.applyTool(o)
	}
}

func (g *Graph) applyTool(o model.Observation) {
	id := model.ToolCallID(o.RunID, o.Correlation.ToolUseID)
	nodeType := model.NodeToolCall
	if name, _ := o.Attrs["name"].(string); isMCP(name) {
		nodeType = model.NodeMCPCall
	}
	n := g.node(id, o.RunID, nodeType)
	g.stamp(n, o)
	if name, ok := o.Attrs["name"].(string); ok && n.Name == "" {
		n.Name = name
	}
	if s, ok := o.Attrs["status"].(string); ok {
		n.Status = model.Status(s)
	}
	mergePayload(n, o.Payload)
	if o.Correlation.MessageID != "" {
		g.upsertEdge(o.RunID, model.EdgeParentChild, model.AssistantTurnID(o.RunID, o.Correlation.MessageID), id)
	}
}

func (g *Graph) stamp(n *model.Node, o model.Observation) {
	if n.TStart == nil || o.ObservedAt.Before(*n.TStart) {
		ts := o.ObservedAt
		n.TStart = &ts
	}
	n.Sources = append(n.Sources, model.SourceRef{Source: o.Source, ObsID: o.ObsID, ObservedAt: o.ObservedAt})
}

func mergePayload(n *model.Node, p *model.Payload) {
	if p == nil {
		return
	}
	if n.Payload == nil {
		n.Payload = &model.Payload{}
	}
	if len(p.Input) > 0 {
		n.Payload.Input = p.Input
	}
	if len(p.Output) > 0 {
		n.Payload.Output = p.Output
	}
	n.Payload.Hash = model.HashPayload(n.Payload)
	n.PayloadHash = n.Payload.Hash
}

func applyTokens(n *model.Node, attrs map[string]any) {
	if v, ok := toInt64(attrs["tokens_in"]); ok {
		n.TokensIn = &v
	}
	if v, ok := toInt64(attrs["tokens_out"]); ok {
		n.TokensOut = &v
	}
}

func toInt64(v any) (int64, bool) {
	switch x := v.(type) {
	case int64:
		return x, true
	case int:
		return int64(x), true
	case float64:
		return int64(x), true
	default:
		return 0, false
	}
}

func isMCP(name string) bool {
	return strings.HasPrefix(name, "mcp__")
}
```

```go
package store

import (
	"database/sql"
	"encoding/json"
	"fmt"

	"github.com/realkarych/catacomb/model"
	_ "modernc.org/sqlite"
)

type sqliteStore struct {
	db      *sql.DB
	marshal func(any) ([]byte, error)
}

const schema = `
CREATE TABLE IF NOT EXISTS observations (obs_id TEXT PRIMARY KEY, run_id TEXT, seq INTEGER, body TEXT);
CREATE TABLE IF NOT EXISTS nodes (id TEXT PRIMARY KEY, run_id TEXT, body TEXT);
CREATE TABLE IF NOT EXISTS edges (id TEXT PRIMARY KEY, run_id TEXT, body TEXT);
CREATE INDEX IF NOT EXISTS idx_nodes_run ON nodes(run_id);
CREATE INDEX IF NOT EXISTS idx_edges_run ON edges(run_id);
`

func OpenSQLite(path string) (Store, error) {
	return openSQLite(sql.Open, path)
}

func openSQLite(open func(driver, dsn string) (*sql.DB, error), path string) (Store, error) {
	db, err := open("sqlite", path)
	if err != nil {
		return nil, fmt.Errorf("store.OpenSQLite: %w", err)
	}
	if _, err := db.Exec(schema); err != nil {
		return nil, fmt.Errorf("store.OpenSQLite schema: %w", err)
	}
	return &sqliteStore{db: db, marshal: json.Marshal}, nil
}

func (s *sqliteStore) AppendObservation(o model.Observation) error {
	body, err := s.marshal(o)
	if err != nil {
		return fmt.Errorf("store.AppendObservation: %w", err)
	}
	_, err = s.db.Exec(
		`INSERT OR REPLACE INTO observations(obs_id, run_id, seq, body) VALUES(?,?,?,?)`,
		o.ObsID, o.RunID, o.Seq, string(body))
	return err
}

func (s *sqliteStore) UpsertNode(n *model.Node) error {
	body, err := s.marshal(n)
	if err != nil {
		return fmt.Errorf("store.UpsertNode: %w", err)
	}
	_, err = s.db.Exec(
		`INSERT INTO nodes(id, run_id, body) VALUES(?,?,?) ON CONFLICT(id) DO UPDATE SET body=excluded.body`,
		n.ID, n.RunID, string(body))
	return err
}

func (s *sqliteStore) UpsertEdge(e *model.Edge) error {
	body, err := s.marshal(e)
	if err != nil {
		return fmt.Errorf("store.UpsertEdge: %w", err)
	}
	_, err = s.db.Exec(
		`INSERT INTO edges(id, run_id, body) VALUES(?,?,?) ON CONFLICT(id) DO UPDATE SET body=excluded.body`,
		e.ID, e.RunID, string(body))
	return err
}

func (s *sqliteStore) Close() error { return s.db.Close() }
```

```go
import (
```

```go
	"io"

	"github.com/realkarych/catacomb/model"

func Snapshot(w io.Writer, nodes []*model.Node, edges []*model.Edge) error {
	enc := json.NewEncoder(w)
	for _, n := range nodes {
		if err := enc.Encode(struct {
			Kind string `json:"kind"`
			*model.Node
		}{"node", n}); err != nil {
			return err
		}

	for _, e := range edges {
		if err := enc.Encode(struct {
			Kind string `json:"kind"`
			*model.Edge
		}{"edge", e}); err != nil {
			return err
		}
	}
	return nil
}
```

```go
package main

import (
	"fmt"
	"os"

	xjsonl "github.com/realkarych/catacomb/export/jsonl"
	ijsonl "github.com/realkarych/catacomb/ingest/jsonl"
	"github.com/realkarych/catacomb/model"
	"github.com/realkarych/catacomb/reduce"
	"github.com/realkarych/catacomb/store"
	"github.com/spf13/cobra"
)

type replayArgs struct {
	input      string
	dbPath     string
	exportPath string
}

func newReplayCmd() *cobra.Command {
	args := replayArgs{}
	cmd := &cobra.Command{
		Use:   "replay <transcript.jsonl>",
		Short: "Build a graph from a recorded Claude Code transcript",
		Args:  cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, positional []string) error {
			args.input = positional[0]
			g, err := runReplay(args)
			if err != nil {
				return err
			}
			cmd.Printf("replayed %s -> %d nodes, %d edges\n", args.input, len(g.Nodes), len(g.Edges))
			return nil
		},
	}
	cmd.Flags().StringVar(&args.dbPath, "db", "catacomb.db", "SQLite database path")
	cmd.Flags().StringVar(&args.exportPath, "export-jsonl", "", "also write a JSONL graph snapshot")
	return cmd
}

func runReplay(args replayArgs) (*reduce.Graph, error) {
	f, err := os.Open(args.input)
	if err != nil {
		return nil, fmt.Errorf("replay open: %w", err)
	}
	defer func() { _ = f.Close() }()

	obs, err := ijsonl.ParseReader(f)
	if err != nil {
		return nil, err
	}

	g := reduce.NewGraph()
	g.ApplyAll(obs)

	if err := persist(args.dbPath, obs, g); err != nil {
		return nil, err
	}
	if args.exportPath != "" {
		if err := export(args.exportPath, g); err != nil {
			return nil, err
		}
	}
	return g, nil
}

func persist(dbPath string, obs []model.Observation, g *reduce.Graph) error {
	s, err := store.OpenSQLite(dbPath)
	if err != nil {
		return err
	}
	defer func() { _ = s.Close() }()

	for _, o := range obs {
		if err := s.AppendObservation(o); err != nil {
			return err
		}
	}
	for _, n := range g.Nodes {
		if err := s.UpsertNode(n); err != nil {
			return err
		}
	}
	for _, e := range g.Edges {
		if err := s.UpsertEdge(e); err != nil {
			return err
		}
	}
	return nil
}

func export(path string, g *reduce.Graph) error {
	out, err := os.Create(path)
	if err != nil {
		return fmt.Errorf("replay export: %w", err)
	}
	defer func() { _ = out.Close() }()

	nodes := make([]*model.Node, 0, len(g.Nodes))
	for _, n := range g.Nodes {
		nodes = append(nodes, n)
	}
	edges := make([]*model.Edge, 0, len(g.Edges))
	for _, e := range g.Edges {
		edges = append(edges, e)
	}
	return xjsonl.Snapshot(out, nodes, edges)
}
```

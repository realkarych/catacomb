# Catacomb M0.1 — Offline Core & Replay — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the transport-agnostic core that turns a recorded Claude Code session (transcript JSONL) into a canonical execution graph, persists it to SQLite, and exports it to JSONL — all offline and unit-testable.

**Architecture:** A reusable core library (`model`, `ingest/jsonl`, `reduce`, `store`, `export/jsonl`) plus a thin `cmd/catacomb` CLI. Data flows transcript file → JSONL parser → `[]Observation` → reducer (canonical-entity merge) → in-memory `Graph` → SQLite store + JSONL exporter. The reducer is a deterministic reduction of an append-only observation stream, so the same fixtures always yield the same golden graph (the basis of every test).

**Tech Stack:** Go 1.23, `github.com/spf13/cobra` (CLI), `modernc.org/sqlite` (pure-Go SQLite, no cgo), `github.com/oklog/ulid/v2` (sortable ids), `github.com/google/go-cmp` (test assertions).

## Global Constraints

- **Language/version:** Go **1.23+**.
- **Module path:** `github.com/realkarych/catacomb`.
- **No cgo:** all dependencies pure-Go so the daemon/CLI cross-compiles to a single static binary. (SQLite via `modernc.org/sqlite`, never `mattn/go-sqlite3`.)
- **License:** Apache-2.0 (repo `LICENSE`); no per-file headers required.
- **Core is transport-agnostic:** packages `model`, `ingest/*`, `reduce`, `store`, `export/*` must not import the CLI, a daemon, or any network/process code. They are importable by embedders.
- **Canonical graph = deterministic reduction** of the observation stream: reducing the same observations in any order yields the same graph (tested in Task 5).
- **Source precedence (M0.1 sources = `jsonl` only; full table in spec §7):** later observations enrich; for a field already set, a higher-precedence source overwrites, else ties break by latest `ObservedAt`.

```go
package model

import (
	"crypto/sha256"
	"encoding/hex"
)

// HashPayload returns the sha256 hex of input||output. Nil payload -> "".
func HashPayload(p *Payload) string {
	if p == nil {
		return ""
	}
	h := sha256.New()
	h.Write(p.Input)
	h.Write(p.Output)
	return hex.EncodeToString(h.Sum(nil))
}
```

```go
// Package jsonl parses Claude Code transcript JSONL into model.Observations.
package jsonl

import (
	"bufio"
	"encoding/json"
	"fmt"
	"io"
	"time"

	"github.com/realkarych/catacomb/model"
)

type line struct {
	Type       string          `json:"type"`
	UUID       string          `json:"uuid"`
	ParentUUID string          `json:"parentUuid"`
	SessionID  string          `json:"sessionId"`
	Timestamp  string          `json:"timestamp"`
	Message    json.RawMessage `json:"message"`
}

type message struct {
	Role    string          `json:"role"`
	ID      string          `json:"id"`
	Model   string          `json:"model"`
	Content json.RawMessage `json:"content"` // string OR []block
	Usage   *usage          `json:"usage"`
}

type usage struct {
	InputTokens  int64 `json:"input_tokens"`
	OutputTokens int64 `json:"output_tokens"`
}

type block struct {
	Type      string          `json:"type"`
	ID        string          `json:"id"`
	Name      string          `json:"name"`
	Input     json.RawMessage `json:"input"`
	ToolUseID string          `json:"tool_use_id"`
	Content   json.RawMessage `json:"content"`
	IsError   bool            `json:"is_error"`
}

func ParseReader(r io.Reader) ([]model.Observation, error) {
	var out []model.Observation
	sc := bufio.NewScanner(r)
	sc.Buffer(make([]byte, 0, 1024*1024), 16*1024*1024)
	var seq uint64
	for sc.Scan() {
		b := sc.Bytes()
		if len(b) == 0 {
			continue
		}
		obs, err := ParseLine(b, seq)
		if err != nil {
			return nil, err
		}
		seq += uint64(len(obs))
		out = append(out, obs...)
	}
	return out, sc.Err()
}

func ParseLine(raw []byte, seq uint64) ([]model.Observation, error) {
	var ln line
	if err := json.Unmarshal(raw, &ln); err != nil {
		return nil, fmt.Errorf("decode line: %w", err)
	}
	ts, _ := time.Parse(time.RFC3339, ln.Timestamp)
	base := func(kind string) model.Observation {
		return model.Observation{
			ObsID:       fmt.Sprintf("obs-%d", seq),
			RunID:       ln.SessionID,
			Source:      model.SourceJSONL,
			Kind:        kind,
			Correlation: model.Correlation{SessionID: ln.SessionID, UUID: ln.UUID},
			ObservedAt:  ts,
			Seq:         seq,
		}
	}
	if len(ln.Message) == 0 {
		return nil, nil
	}
	var msg message
	if err := json.Unmarshal(ln.Message, &msg); err != nil {
		return nil, fmt.Errorf("decode message: %w", err)
	}

	// content is either a JSON string (plain text) or an array of blocks.
	var text string
	var blocks []block
	if err := json.Unmarshal(msg.Content, &text); err != nil {
		if err := json.Unmarshal(msg.Content, &blocks); err != nil {
			return nil, fmt.Errorf("decode content: %w", err)
		}
	}

	var out []model.Observation
	switch ln.Type {
	case "user":
		if text != "" {
			o := base("user_prompt")
			seq++
			o.Seq = seq - 1
			o.ObsID = fmt.Sprintf("obs-%d", o.Seq)
			out = append(out, o)
		}
		for _, bl := range blocks {
			if bl.Type != "tool_result" {
				continue
			}
			o := base("tool_result")
			o.Seq = seq + uint64(len(out))
			o.ObsID = fmt.Sprintf("obs-%d", o.Seq)
			o.Correlation.ToolUseID = bl.ToolUseID
			status := model.StatusOK
			if bl.IsError {
				status = model.StatusError
			}
			o.Attrs = map[string]any{"status": string(status)}
			o.Payload = &model.Payload{Output: bl.Content}
			o.Payload.Hash = model.HashPayload(o.Payload)
			out = append(out, o)
		}
	case "assistant":
		turn := base("assistant_turn")
		turn.Seq = seq + uint64(len(out))
		turn.ObsID = fmt.Sprintf("obs-%d", turn.Seq)
		turn.Correlation.MessageID = msg.ID
		turn.Attrs = map[string]any{"model": msg.Model}
		if msg.Usage != nil {
			turn.Attrs["tokens_in"] = msg.Usage.InputTokens
			turn.Attrs["tokens_out"] = msg.Usage.OutputTokens
		}
		out = append(out, turn)
		for _, bl := range blocks {
			if bl.Type != "tool_use" {
				continue
			}
			o := base("assistant_tool_use")
			o.Seq = seq + uint64(len(out))
			o.ObsID = fmt.Sprintf("obs-%d", o.Seq)
			o.Correlation.ToolUseID = bl.ID
			o.Correlation.MessageID = msg.ID
			o.Attrs = map[string]any{"name": bl.Name}
			o.Payload = &model.Payload{Input: bl.Input}
			o.Payload.Hash = model.HashPayload(o.Payload)
			out = append(out, o)
		}
	}
	return out, nil
}
```

```go
// Package reduce turns an observation stream into a canonical graph.
package reduce

import "github.com/realkarych/catacomb/model"

type Graph struct {
	Nodes map[string]*model.Node
	Edges map[string]*model.Edge
}

func NewGraph() *Graph {
	return &Graph{Nodes: map[string]*model.Node{}, Edges: map[string]*model.Edge{}}
}

func (g *Graph) upsertEdge(runID string, t model.EdgeType, src, dst string) {
	if src == "" || dst == "" {
		return
	}
	id := model.EdgeID(runID, t, src, dst)
	if _, ok := g.Edges[id]; !ok {
		g.Edges[id] = &model.Edge{ID: id, RunID: runID, Type: t, Src: src, Dst: dst}
	}
}

// node returns the existing node or creates it with id/type/run set.
func (g *Graph) node(id, runID string, t model.NodeType) *model.Node {
	n, ok := g.Nodes[id]
	if !ok {
		n = &model.Node{ID: id, RunID: runID, Type: t, Verbosity: "core"}
		g.Nodes[id] = n
	}
	return n
}
```

```go
package reduce

import "github.com/realkarych/catacomb/model"

func (g *Graph) ApplyAll(obs []model.Observation) {
	for _, o := range obs {
		g.Apply(o)
	}
}

func (g *Graph) Apply(o model.Observation) {
	switch o.Kind {
	case "user_prompt":
		n := g.node(model.UserPromptID(o.RunID, o.Correlation.UUID), o.RunID, model.NodeUserPrompt)
		g.stamp(n, o)
		g.upsertEdge(o.RunID, model.EdgeParentChild, model.SessionNodeID(o.RunID), n.ID)

	case "assistant_turn":
		n := g.node(model.AssistantTurnID(o.RunID, o.Correlation.MessageID), o.RunID, model.NodeAssistantTurn)
		g.stamp(n, o)
		applyTokens(n, o.Attrs)

	case "assistant_tool_use", "tool_result":
		id := model.ToolCallID(o.RunID, o.Correlation.ToolUseID)
		t := model.NodeToolCall
		if name, _ := o.Attrs["name"].(string); isMCP(name) {
			t = model.NodeMCPCall
		}
		n := g.node(id, o.RunID, t)
		g.stamp(n, o)
		if name, ok := o.Attrs["name"].(string); ok && n.Name == "" {
			n.Name = name
		}
		if s, ok := o.Attrs["status"].(string); ok {
			n.Status = model.Status(s)
		}
		mergePayload(n, o.Payload)
		if o.Correlation.MessageID != "" {
			g.upsertEdge(o.RunID, model.EdgeParentChild, model.AssistantTurnID(o.RunID, o.Correlation.MessageID), id)
		}
	}
	// session node always exists as the run root.
	g.node(model.SessionNodeID(o.RunID), o.RunID, model.NodeSession)
}

// stamp records timing + provenance common to every observation.
func (g *Graph) stamp(n *model.Node, o model.Observation) {
	if n.TStart == nil || o.ObservedAt.Before(*n.TStart) {
		ts := o.ObservedAt
		n.TStart = &ts
	}
	n.Sources = append(n.Sources, model.SourceRef{Source: o.Source, ObsID: o.ObsID, ObservedAt: o.ObservedAt})
}

func mergePayload(n *model.Node, p *model.Payload) {
	if p == nil {
		return
	}
	if n.Payload == nil {
		n.Payload = &model.Payload{}
	}
	if len(p.Input) > 0 {
		n.Payload.Input = p.Input
	}
	if len(p.Output) > 0 {
		n.Payload.Output = p.Output
	}
	n.Payload.Hash = model.HashPayload(n.Payload)
	n.PayloadHash = n.Payload.Hash
}

func applyTokens(n *model.Node, attrs map[string]any) {
	if v, ok := toInt64(attrs["tokens_in"]); ok {
		n.TokensIn = &v
	}
	if v, ok := toInt64(attrs["tokens_out"]); ok {
		n.TokensOut = &v
	}
}

func toInt64(v any) (int64, bool) {
	switch x := v.(type) {
	case int64:
		return x, true
	case int:
		return int64(x), true
	case float64:
		return int64(x), true
	}
	return 0, false
}

func isMCP(name string) bool {
	return len(name) > 5 && name[:5] == "mcp__"
}
```

```go
package store

import (
	"database/sql"
	"encoding/json"

	"github.com/realkarych/catacomb/model"
	_ "modernc.org/sqlite"
)

type sqliteStore struct{ db *sql.DB }

const schema = `
CREATE TABLE IF NOT EXISTS observations (
  obs_id TEXT PRIMARY KEY, run_id TEXT, seq INTEGER, body TEXT
);
CREATE TABLE IF NOT EXISTS nodes (
  id TEXT PRIMARY KEY, run_id TEXT, body TEXT
);
CREATE TABLE IF NOT EXISTS edges (
  id TEXT PRIMARY KEY, run_id TEXT, body TEXT
);
CREATE INDEX IF NOT EXISTS idx_nodes_run ON nodes(run_id);
CREATE INDEX IF NOT EXISTS idx_edges_run ON edges(run_id);
`

func OpenSQLite(path string) (Store, error) {
	db, err := sql.Open("sqlite", path)
	if err != nil {
		return nil, err
	}
	if _, err := db.Exec(schema); err != nil {
		db.Close()
		return nil, err
	}
	return &sqliteStore{db: db}, nil
}

func (s *sqliteStore) AppendObservation(o model.Observation) error {
	b, err := json.Marshal(o)
	if err != nil {
		return err
	}
	_, err = s.db.Exec(
		`INSERT OR REPLACE INTO observations(obs_id,run_id,seq,body) VALUES(?,?,?,?)`,
		o.ObsID, o.RunID, o.Seq, string(b))
	return err
}

func (s *sqliteStore) UpsertNode(n *model.Node) error {
	b, err := json.Marshal(n)
	if err != nil {
		return err
	}
	_, err = s.db.Exec(
		`INSERT INTO nodes(id,run_id,body) VALUES(?,?,?)
		 ON CONFLICT(id) DO UPDATE SET body=excluded.body`,
		n.ID, n.RunID, string(b))
	return err
}

func (s *sqliteStore) UpsertEdge(e *model.Edge) error {
	b, err := json.Marshal(e)
	if err != nil {
		return err
	}
	_, err = s.db.Exec(
		`INSERT INTO edges(id,run_id,body) VALUES(?,?,?)
		 ON CONFLICT(id) DO UPDATE SET body=excluded.body`,
		e.ID, e.RunID, string(b))
	return err
}

func (s *sqliteStore) LoadRunGraph(runID string) ([]*model.Node, []*model.Edge, error) {
	nodes, err := loadNodes(s.db, runID)
	if err != nil {
		return nil, nil, err
	}
	edges, err := loadEdges(s.db, runID)
	if err != nil {
		return nil, nil, err
	}
	return nodes, edges, nil
}

func loadNodes(db *sql.DB, runID string) ([]*model.Node, error) {
	rows, err := db.Query(`SELECT body FROM nodes WHERE run_id=? ORDER BY id`, runID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var out []*model.Node
	for rows.Next() {
		var body string
		if err := rows.Scan(&body); err != nil {
			return nil, err
		}
		var n model.Node
		if err := json.Unmarshal([]byte(body), &n); err != nil {
			return nil, err
		}
		out = append(out, &n)
	}
	return out, rows.Err()
}

func loadEdges(db *sql.DB, runID string) ([]*model.Edge, error) {
	rows, err := db.Query(`SELECT body FROM edges WHERE run_id=? ORDER BY id`, runID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var out []*model.Edge
	for rows.Next() {
		var body string
		if err := rows.Scan(&body); err != nil {
			return nil, err
		}
		var e model.Edge
		if err := json.Unmarshal([]byte(body), &e); err != nil {
			return nil, err
		}
		out = append(out, &e)
	}
	return out, rows.Err()
}

func (s *sqliteStore) Close() error { return s.db.Close() }
```

```go
package main

import (
	"os"

	xjsonl "github.com/realkarych/catacomb/export/jsonl"
	ijsonl "github.com/realkarych/catacomb/ingest/jsonl"
	"github.com/realkarych/catacomb/model"

	"github.com/spf13/cobra"

func newReplayCmd() *cobra.Command {
	var dbPath, exportPath string
	cmd := &cobra.Command{
		Use:   "replay <transcript.jsonl>",
		Short: "Build a graph from a recorded Claude Code transcript",
		Args:  cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			f, err := os.Open(args[0])
			if err != nil {
				return err
			}
			defer f.Close()

			obs, err := ijsonl.ParseReader(f)
			if err != nil {
				return err
			}
			g := reduce.NewGraph()
			g.ApplyAll(obs)

			s, err := store.OpenSQLite(dbPath)
			if err != nil {
				return err
			}
			defer s.Close()
			for _, o := range obs {
				if err := s.AppendObservation(o); err != nil {
					return err
				}
			}
			for _, n := range g.Nodes {
				if err := s.UpsertNode(n); err != nil {
					return err
				}
			}
			for _, e := range g.Edges {
				if err := s.UpsertEdge(e); err != nil {
					return err
				}
			}

			if exportPath != "" {
				out, err := os.Create(exportPath)
				if err != nil {
					return err
				}
				defer out.Close()
				nodes, edges := mapsToSlices(g)
				if err := xjsonl.Snapshot(out, nodes, edges); err != nil {
					return err
				}
			}
			cmd.Printf("replayed %d observations -> %d nodes, %d edges\n", len(obs), len(g.Nodes), len(g.Edges))
			return nil
		},
	}
	cmd.Flags().StringVar(&dbPath, "db", "catacomb.db", "SQLite database path")
	cmd.Flags().StringVar(&exportPath, "export-jsonl", "", "also write a JSONL graph snapshot")
	return cmd
}

func mapsToSlices(g *reduce.Graph) ([]*model.Node, []*model.Edge) {
	nodes := make([]*model.Node, 0, len(g.Nodes))
	for _, n := range g.Nodes {
		nodes = append(nodes, n)
	}
	edges := make([]*model.Edge, 0, len(g.Edges))
	for _, e := range g.Edges {
		edges = append(edges, e)
	}
	return nodes, edges
}
```

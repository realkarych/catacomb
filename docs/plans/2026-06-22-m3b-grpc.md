# M3b — gRPC Streaming Realtime Surface Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development`
> (recommended) or `superpowers:executing-plans` to implement this plan task-by-task.
> Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `GraphService.Subscribe` gRPC server-streaming endpoint that delivers the same
race-free snapshot-then-live-delta feed as the M3a SSE surface, protected by a streaming bearer
interceptor, driven by the existing `SubscribeFiltered`/`Unsubscribe`/`matchDelta` core.

**Architecture:** Task 1 is pure infrastructure — create `proto/catacomb/v1/graph.proto`, wire
`buf.yaml`/`buf.gen.yaml`, add a `make proto` Makefile target (which `go install`s the two
code-gen plugins at pinned versions then runs `buf generate`), exclude `gen/` from the coverage
gate, and commit the generated `gen/catacombv1/*.pb.go`. Task 2 is the hand-written Go logic:
`streamBearerInterceptor`, `graphServer.Subscribe`, `toProtoDelta` (nilsPayload, keeps
`PayloadHash`), and registration in `newGRPCServer` — all fully TDD-tested via an in-memory
`bufconn` listener. Task 3 (optional, deferrable to M5) adds the `catacomb watch --protocol grpc`
CLI client.

**Tech Stack:** Go 1.26; `buf` v2 + `protoc-gen-go@v1.36.11` + `protoc-gen-go-grpc@v1.5.1`;
`google.golang.org/protobuf v1.36.11` + `google.golang.org/grpc v1.81.1` (already in `go.mod`);
`google.golang.org/grpc/test/bufconn` for in-memory gRPC tests; `testify/require`.

## Global Constraints

- Go 1.26 pure-Go (no cgo).
- NO comments except `//go:build`, `//go:embed`, `//go:generate` and the generated-file header
  (`internal/codepolicy` enforces this; the `// Code generated … DO NOT EDIT.` header in
  `gen/catacombv1/` is auto-skipped by codepolicy, golangci-lint, and coverage gate).
- 100% line coverage under `-race` (`make cover`) with generated `*.pb.go` excluded from the gate
  (add `- ^gen/` to `.testcoverage.yml` `exclude.paths`).
- golangci-lint v2 clean: gofumpt, goimports local-prefix `github.com/realkarych/catacomb`,
  govet shadow, **forbidigo bans `time.Sleep`** — use tickers/channels; `unparam`; `errcheck`;
  `rowserrcheck`; `bodyclose`. Generated files auto-skipped via `generated: strict`.
- Never `go mod tidy`; add deps via `go get`.
- Single-mutex daemon (`d.mu`) — the subscription handshake holds it; the per-connection stream
  loop does NOT hold it.
- Cross-platform: `GOOS=windows go build ./...` must be clean; no unix-only syscalls.
- Tests use `require.Eventually`; close listeners before temp-dir cleanup.
- Loopback + bearer trust boundary (ADR-0013).
- One commit per task; never commit to master mid-plan.

package catacomb.v1;

service GraphService {
  rpc Subscribe(SubscribeRequest) returns (stream GraphDelta);

message SubscribeRequest {
  string run_id = 1;
  repeated string node_types = 2;
  repeated string tiers = 3;

  string subagent_type = 7;

func bearerInterceptor(token string) grpc.UnaryServerInterceptor {
    want := []byte("Bearer " + token)
    return func(ctx context.Context, req any, _ *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (any, error) {
        md, ok := metadata.FromIncomingContext(ctx)
        if !ok {
            return nil, status.Error(codes.Unauthenticated, "missing metadata")
        }
        vals := md.Get("authorization")
        if len(vals) == 0 || subtle.ConstantTimeCompare([]byte(vals[0]), want) != 1 {
            return nil, status.Error(codes.Unauthenticated, "invalid token")
        }
        return handler(ctx, req)
    }
}

func streamBearerInterceptor(token string) grpc.StreamServerInterceptor {
    want := []byte("Bearer " + token)
    return func(srv any, ss grpc.ServerStream, _ *grpc.StreamServerInfo, handler grpc.StreamHandler) error {
        md, ok := metadata.FromIncomingContext(ss.Context())
        if !ok {
            return status.Error(codes.Unauthenticated, "missing metadata")
        }
        vals := md.Get("authorization")
        if len(vals) == 0 || subtle.ConstantTimeCompare([]byte(vals[0]), want) != 1 {
            return status.Error(codes.Unauthenticated, "invalid token")
        }
        return handler(srv, ss)
    }
}

func (g *graphServer) Subscribe(req *catacombv1.SubscribeRequest, stream catacombv1.GraphService_SubscribeServer) error {
    f := SubFilter{
        RunID:     req.GetRunId(),
        NodeTypes: req.GetNodeTypes(),
        Tiers:     req.GetTiers(),
    }
    sub := g.d.SubscribeFiltered(f, subBufSize)
    defer g.d.Unsubscribe(sub)
    for _, snap := range sub.Snapshot {
        if err := stream.Send(toProtoDelta(snap)); err != nil {
            return err
        }
    }

        case delta, ok := <-sub.Consumer.C:

func encodeAttrs(attrs map[string]any) map[string]string {
    out := make(map[string]string, len(attrs))
    for k, v := range attrs {
        b, err := json.Marshal(v)

    srv := grpc.NewServer(

        grpc.StreamInterceptor(streamBearerInterceptor(token)),

    catacombv1.RegisterGraphServiceServer(srv, &graphServer{d: d})

func (d *Daemon) serveGRPC(
    ctx context.Context,
    srv *grpc.Server,
    ln net.Listener,
    serveFn func(*grpc.Server, net.Listener) error,
    waitFn func(ctx context.Context, d time.Duration) bool,
) {
    backoff := grpcBackoffBase
    for {
        if ctx.Err() != nil {
            return
        }
        var serveErr error
        func() {
            defer func() {
                if r := recover(); r != nil {
                    serveErr = status.Errorf(codes.Internal, "panic: %v", r)
                }
            }()
            serveErr = serveFn(srv, ln)
        }()
        if ctx.Err() != nil {
            return
        }
        if serveErr != nil {
            if !waitFn(ctx, backoff) {
                return
            }
            if backoff*grpcBackoffFactor < grpcBackoffCap {
                backoff *= grpcBackoffFactor
            } else {
                backoff = grpcBackoffCap
            }
        }
    }
}

func runWatchGRPC(ctx context.Context, out io.Writer, addr, token, runID string, nodeTypes, tiers []string) error {
    conn, err := grpc.NewClient(addr,
        grpc.WithTransportCredentials(insecure.NewCredentials()),

    }
    defer conn.Close()
    client := catacombv1.NewGraphServiceClient(conn)

        Tiers:     tiers,

    }
    enc := json.NewEncoder(out)
    for {
        delta, recvErr := stream.Recv()
        if recvErr != nil {
            if ctx.Err() != nil {
                return nil
            }
            return recvErr
        }
        if encErr := enc.Encode(delta); encErr != nil {
            return encErr
        }
    }
}
